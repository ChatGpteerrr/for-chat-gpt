local runService = game:GetService("RunService")
local workspaceService = game:GetService("Workspace")

local stretchWidthScale = 0.92
local stretchHeightScale = 0.58

if getgenv().StretchScreen and type(getgenv().StretchScreen.disable) == "function" then
    pcall(function()
        getgenv().StretchScreen.disable()
    end)
end

local stretchConnection = nil
local stretchEnabled = false

local stretchCFrameOffset = (function(widthScale, heightScale)
    return CFrame.new(0, 0, 0, widthScale, 0, 0, 0, heightScale, 0, 0, 0, 1)
end)(stretchWidthScale, stretchHeightScale)

local function applyStretchToCamera()
    local currentCamera = workspaceService.CurrentCamera
    if currentCamera then
        pcall(function()
            currentCamera.CFrame = currentCamera.CFrame * stretchCFrameOffset
        end)
    end
end

local function enableStretchScreen()
    if not stretchEnabled then
        stretchEnabled = true
        stretchConnection = runService.RenderStepped:Connect(function()
            if workspaceService.CurrentCamera then
                applyStretchToCamera()
            end
        end)
    end
end

local function disableStretchScreen()
    if stretchEnabled then
        stretchEnabled = false
        if stretchConnection and stretchConnection.Connected then
            pcall(function()
                stretchConnection:Disconnect()
            end)
        end
        stretchConnection = nil
    end
end

local function toggleStretchScreen()
    if stretchEnabled then
        disableStretchScreen()
    else
        enableStretchScreen()
    end
end

getgenv().StretchScreen = getgenv().StretchScreen or {}
getgenv().StretchScreen.enable = enableStretchScreen
getgenv().StretchScreen.disable = disableStretchScreen
getgenv().StretchScreen.toggle = toggleStretchScreen
getgenv().StretchScreen.isEnabled = function()
    return stretchEnabled
end

getgenv().StretchScreen._internal = {
    widthScale = stretchWidthScale,
    heightScale = stretchHeightScale
}

        uiLibrary:Notify({
            Title = "Merebennie Hub",

            Duration = 3

        local localPlayer = game:GetService("Players").LocalPlayer

        local function disableCharacterShake()
            pcall(function()
                local character = localPlayer.Character or localPlayer.CharacterAdded:Wait()
                if character then
                    local success, characterClient = pcall(function()
                        return character:WaitForChild("CharacterHandler", 5):WaitForChild("Client", 5)
                    end)

                    if success and characterClient then
                        local clientEnv = getsenv(characterClient)
                        if clientEnv and clientEnv.shared and clientEnv.shared.addshake then
                            pcall(function()
                                hookfunction(clientEnv.shared.addshake, function()
                                    return nil
                                end)
                            end)
                        end
                    end
                end
            end)
        end

        disableCharacterShake()

        if not _G.__Merebennie_ShakeHook_Connected then
            _G.__Merebennie_ShakeHook_Connected = true

            localPlayer.CharacterAdded:Connect(function()
                task.wait(0.8)
                disableCharacterShake()
            end)
        end

        uiLibrary:Notify({
            Title = "Success",
            Content = "Shakes disabled (will persist after respawn).",
            Duration = 4
        })
    end

optimizationTab:CreateButton({
    Name = "Fps Booster + Better lighting",
    Callback = function()
        uiLibrary:Notify({
            Title = "Merebennie Hub",
            Content = "Applying FPS Booster + Gray lighting...",

        })

        pcall(function()
            local playersService = game:GetService("Players")
            local lightingService = game:GetService("Lighting")
            local workspaceRef = game:GetService("Workspace")
            game:GetService("RunService")

            local autoApplyBooster = true
            local grayColor = Color3.fromRGB(140, 140, 140)
            local useFog = true
            local fogStart = 0
            local fogEnd = 1000000
            local removeTextures = false
            local revertEnvironmentOnRestore = true
            local hideLocalPlayerHead = true

            local localPlayer = playersService.LocalPlayer

            if not localPlayer then
                for _ = 1, 20 do
                    task.wait(0.03)
                    localPlayer = playersService.LocalPlayer or localPlayer
                    if localPlayer then
                        break
                    end
                end
            end

            local environmentBackup = {
                lighting = {},
                postEffects = {},
                particles = {},
                trails = {},
                beams = {},
                lights = {},
                decals = {},
                skies = {},
                atmospheres = {},
                createdSky = nil,
                createdAtmosphere = nil,
                playerVisuals = {},
                removedInstances = {}
            }

            local function safeGet(propertyFunction, ...)
                local success, value = pcall(propertyFunction, ...)
                if success then
                    return value
                end
                return nil
            end

            local function saveCurrentLighting()
                local lighting = lightingService

                environmentBackup.lighting.Ambient = safeGet(function()
                    return lighting.Ambient
                end)

                environmentBackup.lighting.OutdoorAmbient = safeGet(function()
                    return lighting.OutdoorAmbient
                end)

                environmentBackup.lighting.FogColor = safeGet(function()
                    return lighting.FogColor
                end)

                environmentBackup.lighting.FogStart = safeGet(function()
                    return lighting.FogStart
                end)

                environmentBackup.lighting.FogEnd = safeGet(function()
                    return lighting.FogEnd
                end)

                environmentBackup.lighting.GlobalShadows = safeGet(function()
                    return lighting.GlobalShadows
                end)

                environmentBackup.lighting.Brightness = safeGet(function()
                    return lighting.Brightness
                end)

                environmentBackup.lighting.ColorShiftTop = safeGet(function()
                    return lighting.ColorShift_Top
                end)

                environmentBackup.lighting.ColorShiftBottom = safeGet(function()
                    return lighting.ColorShift_Bottom
                end)

                environmentBackup.lighting.ClockTime = safeGet(function()
                    return lighting.ClockTime
                end)
            end

            local function applyGrayLighting()
                saveCurrentLighting()

                safeGet(function()
                    lightingService.Ambient = grayColor
                end)

                safeGet(function()
                    lightingService.OutdoorAmbient = grayColor
                end)

                if useFog then
                    safeGet(function()
                        lightingService.FogColor = grayColor
                        lightingService.FogStart = fogStart
                        lightingService.FogEnd = fogEnd
                    end)
                end

                safeGet(function()
                    lightingService.GlobalShadows = false
                end)

                pcall(function()
                    lightingService.ColorShift_Top = grayColor
                end)

                pcall(function()
                    lightingService.ColorShift_Bottom = grayColor
                end)

                pcall(function()
                    lightingService.Brightness = 1
                end)
            end

            local function removeExistingSkiesAndAtmospheres()
                local lighting = lightingService

                for _, child in ipairs(lighting:GetChildren()) do
                    if child:IsA("Sky") then
                        local success, cloned = pcall(function()
                            return child:Clone()
                        end)
                        if success and cloned then
                            table.insert(environmentBackup.skies, cloned)
                        end
                        pcall(function()
                            child:Destroy()
                        end)
                    end
                end

                for _, descendant in ipairs(workspaceRef:GetDescendants()) do
                    if descendant:IsA("Sky") then
                        local success, cloned = pcall(function()
                            return descendant:Clone()
                        end)
                        if success and cloned then
                            table.insert(environmentBackup.skies, cloned)
                        end
                        pcall(function()
                            descendant:Destroy()
                        end)
                    end
                end

                for _, child in ipairs(lighting:GetChildren()) do
                    if child:IsA("Atmosphere") then
                        local success, cloned = pcall(function()
                            return child:Clone()
                        end)
                        if success and cloned then
                            table.insert(environmentBackup.atmospheres, cloned)
                        end
                        pcall(function()
                            child:Destroy()
                        end)
                    end
                end
            end

                local customAtmosphere = Instance.new("Atmosphere")

                pcall(function()

                    pcall(function()
                        customAtmosphere.Offset = 0
                    end)

                    customAtmosphere.Parent = lightingService
                end)

                environmentBackup.createdAtmosphere = customAtmosphere

                local customSky = Instance.new("Sky")

                pcall(function()
                    customSky.Name = "FPSBooster_Sky"

                    pcall(function()
                        customSky.SkyboxBk = ""
                    end)

                    pcall(function()
                        customSky.SkyboxDn = ""
                    end)

                    pcall(function()
                        customSky.SkyboxFt = ""
                    end)

                    pcall(function()
                        customSky.SkyboxLf = ""
                    end)

                    pcall(function()
                        customSky.SkyboxRt = ""
                    end)

                    pcall(function()
                        customSky.SkyboxUp = ""
                    end)

                    customSky.Parent = lightingService
                end)

                environmentBackup.createdSky = customSky
            end

            local function disablePostEffects()
                for _, effect in ipairs(lightingService:GetChildren()) do
                    if effect ~= nil and effect.ClassName then
                        local hasEnabledProperty, hasEnabled = pcall(function()
                            return effect.Enabled ~= nil
                        end)

                        if hasEnabledProperty and hasEnabled then
                            local originalEnabled = safeGet(function()
                                return effect.Enabled
                            end)
                            if originalEnabled ~= nil then
                                environmentBackup.postEffects[effect] = originalEnabled
                                safeGet(function()
                                    effect.Enabled = false
                                end)
                            end
                        else
                            local originalIntensity = safeGet(function()
                                return effect.Intensity
                            end)
                            if originalIntensity ~= nil then
                                environmentBackup.postEffects[effect] = {
                                    Intensity = originalIntensity
                                }
                                safeGet(function()
                                    effect.Intensity = 0
                                end)
                            end
                        end
                    end
                end
            end

            local function processVisualInstanceGroup(rootInstance)
                for _, descendant in ipairs(rootInstance:GetDescendants()) do
                    if descendant:IsA("ParticleEmitter") then
                        local success, enabledState = pcall(function()
                            return descendant.Enabled
                        end)
                        if success then
                            environmentBackup.particles[descendant] = enabledState
                            pcall(function()
                                descendant.Enabled = false
                            end)
                        end

                        pcall(function()
                            if descendant.GetAttribute and descendant:GetAttribute("FPSBooster_RateSaved") == nil then
                                descendant:SetAttribute("FPSBooster_RateSaved", descendant.Rate)
                            end
                            descendant.Rate = 0
                        end)
                    elseif descendant:IsA("Trail") then
                        local success, enabledState = pcall(function()
                            return descendant.Enabled
                        end)
                        if success then
                            environmentBackup.trails[descendant] = enabledState
                            pcall(function()
                                descendant.Enabled = false
                            end)
                        end
                    elseif descendant:IsA("Beam") then
                        local success, enabledState = pcall(function()
                            return descendant.Enabled
                        end)
                        if success then
                            environmentBackup.beams[descendant] = enabledState
                            pcall(function()
                                descendant.Enabled = false
                            end)
                        end
                    elseif descendant:IsA("Sparkles") or descendant:IsA("Fire") or descendant:IsA("Smoke") then
                        local success, enabledState = pcall(function()
                            return descendant.Enabled
                        end)
                        if success then
                            environmentBackup.particles[descendant] = enabledState
                            pcall(function()
                                descendant.Enabled = false
                            end)
                        end
                    elseif descendant:IsA("PointLight") or descendant:IsA("SurfaceLight") or descendant:IsA("SpotLight") then
                        local success, enabledState = pcall(function()
                            return descendant.Enabled
                        end)
                        if success then
                            environmentBackup.lights[descendant] = enabledState
                            pcall(function()
                                descendant.Enabled = false
                            end)
                        else
                            local brightnessSuccess, brightnessValue = pcall(function()
                                return descendant.Brightness
                            end)
                            if brightnessSuccess then
                                environmentBackup.lights[descendant] = {
                                    Brightness = brightnessValue
                                }
                                pcall(function()
                                    descendant.Brightness = 0
                                end)
                            end
                        end
                    elseif descendant:IsA("Decal") or descendant:IsA("Texture") then
                        local success, transparencyValue = pcall(function()
                            return descendant.Transparency
                        end)
                        if success then
                            environmentBackup.decals[descendant] = transparencyValue

                            if removeTextures then
                                if descendant.Parent then
                                    table.insert(environmentBackup.removedInstances, descendant)
                                end
                                pcall(function()
                                    descendant:Destroy()
                                end)
                            else
                                pcall(function()
                                    descendant.Transparency = 1
                                end)
                            end
                        end
                    elseif descendant:IsA("SurfaceGui") or descendant:IsA("BillboardGui") then
                        pcall(function()
                            descendant.Enabled = false
                        end)
                    end
                end
            end

            local characterVisualConnection = nil

            local function registerPlayerVisualPart(instance)
                if instance and instance.Parent then
                    if instance:IsA("BasePart") then
                        if environmentBackup.playerVisuals[instance] == nil then
                            environmentBackup.playerVisuals[instance] = {
                                Transparency = safeGet(function()
                                    return instance.Transparency
                                end),
                                LocalTransparencyModifier = safeGet(function()
                                    return instance.LocalTransparencyModifier
                                end)
                            }
                        end
                    elseif instance:IsA("Decal") or instance:IsA("Texture") then
                        if environmentBackup.playerVisuals[instance] == nil then
                            environmentBackup.playerVisuals[instance] = {
                                Transparency = safeGet(function()
                                    return instance.Transparency
                                end)
                            }
                        end
                    elseif instance:IsA("Accessory") and environmentBackup.playerVisuals[instance] == nil then
                        environmentBackup.playerVisuals[instance] = {
                            Saved = true
                        }
                    end
                end
            end

            local function hideLocalCharacterVisuals(character)
                if character then
                    for _, descendant in ipairs(character:GetDescendants()) do
                        if descendant:IsA("BasePart") and descendant.Name == "Head" then
                            registerPlayerVisualPart(descendant)
                            pcall(function()
                                descendant.Transparency = 1
                            end)

                            for _, headChild in ipairs(descendant:GetDescendants()) do
                                if headChild:IsA("Decal") then
                                    registerPlayerVisualPart(headChild)
                                    pcall(function()
                                        headChild.Transparency = 1
                                    end)
                                end
                            end
                        end

                        if descendant:IsA("Accessory") then
                            registerPlayerVisualPart(descendant)

                            for _, accessoryChild in ipairs(descendant:GetDescendants()) do
                                if accessoryChild:IsA("BasePart") then
                                    registerPlayerVisualPart(accessoryChild)
                                    pcall(function()
                                        accessoryChild.Transparency = 1
                                    end)
                                elseif accessoryChild:IsA("Decal") or accessoryChild:IsA("Texture") then
                                    registerPlayerVisualPart(accessoryChild)
                                    pcall(function()
                                        accessoryChild.Transparency = 1
                                    end)
                                end
                            end
                        end

                        if descendant:IsA("MeshPart") or descendant:IsA("UnionOperation") or descendant:IsA("Part") then
                            local lowerName = tostring(descendant.Name):lower()
                            local parentName = ""
                            if descendant.Parent then
                                parentName = (tostring(descendant.Parent.Name) or ""):lower()
                            end

                            if parentName == "head" or lowerName:find("hair") or lowerName:find("hat") then
                                registerPlayerVisualPart(descendant)
                                pcall(function()
                                    descendant.Transparency = 1
                                end)
                            end
                        end

                        if descendant:IsA("Decal") and descendant.Name:lower() == "face" then
                            registerPlayerVisualPart(descendant)
                            pcall(function()
                                descendant.Transparency = 1
                            end)
                        end
                    end

                    if characterVisualConnection then
                        pcall(function()
                            characterVisualConnection:Disconnect()
                        end)
                    end

                    characterVisualConnection = character.DescendantAdded:Connect(function(newDescendant)
                        task.wait(0.01)

                        pcall(function()
                            if newDescendant:IsA("BasePart") and newDescendant.Name == "Head" then
                                registerPlayerVisualPart(newDescendant)
                                newDescendant.Transparency = 1
                            elseif newDescendant:IsA("Accessory") then
                                registerPlayerVisualPart(newDescendant)

                                for _, accessoryChild in ipairs(newDescendant:GetDescendants()) do
                                    if accessoryChild:IsA("BasePart") then
                                        registerPlayerVisualPart(accessoryChild)
                                        pcall(function()
                                            accessoryChild.Transparency = 1
                                        end)
                                    elseif accessoryChild:IsA("Decal") or accessoryChild:IsA("Texture") then
                                        registerPlayerVisualPart(accessoryChild)
                                        pcall(function()
                                            accessoryChild.Transparency = 1
                                        end)
                                    end
                                end
                            elseif newDescendant:IsA("Decal") and newDescendant.Name:lower() == "face" then
                                registerPlayerVisualPart(newDescendant)
                                newDescendant.Transparency = 1
                            else
                                local lowerName = newDescendant.Name:lower()
                                local parentName = ""
                                if newDescendant.Parent then
                                    parentName = (newDescendant.Parent.Name or ""):lower()
                                end

                                if parentName == "head" or lowerName:find("hair") or lowerName:find("hat") then
                                    registerPlayerVisualPart(newDescendant)
                                    pcall(function()
                                        newDescendant.Transparency = 1
                                    end)
                                end
                            end
                        end)
                    end)
                end
            end

            local function restoreLocalCharacterVisuals()
                for instance, originalData in pairs(environmentBackup.playerVisuals) do
                    if instance and instance.Parent then
                        pcall(function()
                            if instance:IsA("BasePart") then
                                if originalData.Transparency ~= nil then
                                    instance.Transparency = originalData.Transparency
                                end

                                if originalData.LocalTransparencyModifier ~= nil then
                                    pcall(function()
                                        instance.LocalTransparencyModifier = originalData.LocalTransparencyModifier
                                    end)
                                end
                            elseif (instance:IsA("Decal") or instance:IsA("Texture")) and originalData.Transparency ~= nil then
                                instance.Transparency = originalData.Transparency
                            end
                        end)
                    end
                end

                environmentBackup.playerVisuals = {}

                if characterVisualConnection then
                    pcall(function()
                        characterVisualConnection:Disconnect()
                    end)
                    characterVisualConnection = nil
                end
            end

            local worldDescendantAddedConnection = nil

            local function applyFPSBooster()
                applyGrayLighting()
                removeExistingSkiesAndAtmospheres()
                createCustomAtmosphereAndSky()
                disablePostEffects()
                processVisualInstanceGroup(workspaceRef)

                if localPlayer and localPlayer.Character then
                    processVisualInstanceGroup(localPlayer.Character)
                end

                if hideLocalPlayerHead and localPlayer then
                    if localPlayer.Character then
                        hideLocalCharacterVisuals(localPlayer.Character)
                    end

                    localPlayer.CharacterAdded:Connect(function(newCharacter)
                        task.wait(0.06)
                        hideLocalCharacterVisuals(newCharacter)
                    end)
                end
            end

            local function restoreEnvironmentFromFPSBooster()
                pcall(function()
                    if environmentBackup.lighting.Ambient then
                        lightingService.Ambient = environmentBackup.lighting.Ambient
                    end

                    if environmentBackup.lighting.OutdoorAmbient then
                        lightingService.OutdoorAmbient = environmentBackup.lighting.OutdoorAmbient
                    end

                    if environmentBackup.lighting.FogColor then
                        lightingService.FogColor = environmentBackup.lighting.FogColor
                    end

                    if environmentBackup.lighting.FogStart then
                        lightingService.FogStart = environmentBackup.lighting.FogStart
                    end

                    if environmentBackup.lighting.FogEnd then
                        lightingService.FogEnd = environmentBackup.lighting.FogEnd
                    end

                    if environmentBackup.lighting.GlobalShadows ~= nil then
                        lightingService.GlobalShadows = environmentBackup.lighting.GlobalShadows
                    end

                    if environmentBackup.lighting.Brightness then
                        lightingService.Brightness = environmentBackup.lighting.Brightness
                    end

                    if environmentBackup.lighting.ColorShiftTop then
                        pcall(function()
                            lightingService.ColorShift_Top = environmentBackup.lighting.ColorShiftTop
                        end)
                    end

                    if environmentBackup.lighting.ColorShiftBottom then
                        pcall(function()
                            lightingService.ColorShift_Bottom = environmentBackup.lighting.ColorShiftBottom
                        end)
                    end

                    if environmentBackup.lighting.ClockTime then
                        pcall(function()
                            lightingService.ClockTime = environmentBackup.lighting.ClockTime
                        end)
                    end
                end)

                for effect, originalData in pairs(environmentBackup.postEffects) do
                    if effect and effect.Parent then
                        pcall(function()
                            if type(originalData) ~= "table" or originalData.Intensity == nil then
                                effect.Enabled = originalData
                            else
                                effect.Intensity = originalData.Intensity
                            end
                        end)
                    end
                end

                for particle, originalEnabled in pairs(environmentBackup.particles) do
                    if particle and particle.Parent then
                        pcall(function()
                            particle.Enabled = originalEnabled
                        end)

                        pcall(function()
                            if particle.GetAttribute and particle:GetAttribute("FPSBooster_RateSaved") then
                                particle.Rate = particle:GetAttribute("FPSBooster_RateSaved")
                                particle:SetAttribute("FPSBooster_RateSaved", nil)
                            end
                        end)
                    end
                end

                for trail, originalEnabled in pairs(environmentBackup.trails) do
                    if trail and trail.Parent then
                        pcall(function()
                            trail.Enabled = originalEnabled
                        end)
                    end
                end

                for beam, originalEnabled in pairs(environmentBackup.beams) do
                    if beam and beam.Parent then
                        pcall(function()
                            beam.Enabled = originalEnabled
                        end)
                    end
                end

                for lightInstance, originalData in pairs(environmentBackup.lights) do
                    if lightInstance and lightInstance.Parent then
                        pcall(function()
                            if type(originalData) ~= "table" or not originalData.Brightness then
                                lightInstance.Enabled = originalData
                            else
                                lightInstance.Brightness = originalData.Brightness
                            end
                        end)
                    end
                end

                for decal, originalTransparency in pairs(environmentBackup.decals) do
                    if decal and decal.Parent then
                        pcall(function()
                            decal.Transparency = originalTransparency
                        end)
                    end
                end

                if environmentBackup.createdSky and environmentBackup.createdSky.Parent then
                    pcall(function()
                        environmentBackup.createdSky:Destroy()
                    end)
                end

                if environmentBackup.createdAtmosphere and environmentBackup.createdAtmosphere.Parent then
                    pcall(function()
                        environmentBackup.createdAtmosphere:Destroy()
                    end)
                end

                for _, savedSky in ipairs(environmentBackup.skies) do
                    if savedSky and not savedSky.Parent then
                        pcall(function()
                            savedSky.Parent = lightingService
                        end)
                    end
                end

                for _, savedAtmosphere in ipairs(environmentBackup.atmospheres) do
                    if savedAtmosphere and not savedAtmosphere.Parent then
                        pcall(function()
                            savedAtmosphere.Parent = lightingService
                        end)
                    end
                end

                restoreLocalCharacterVisuals()

                environmentBackup.postEffects = {}
                environmentBackup.particles = {}
                environmentBackup.trails = {}
                environmentBackup.beams = {}
                environmentBackup.lights = {}
                environmentBackup.decals = {}
                environmentBackup.skies = {}
                environmentBackup.atmospheres = {}
                environmentBackup.removedInstances = {}

                if worldDescendantAddedConnection and worldDescendantAddedConnection.Connected then
                    pcall(function()
                        worldDescendantAddedConnection:Disconnect()
                    end)
                end
            end

            getgenv().FPSBooster = getgenv().FPSBooster or {}
            getgenv().FPSBooster.apply = applyFPSBooster
            getgenv().FPSBooster.restore = restoreEnvironmentFromFPSBooster

            local fpsBoosterConfig = {
                GRAY_COLOR = grayColor,
                REMOVE_TEXTURES = removeTextures,
                HIDE_LOCAL_HEAD = hideLocalPlayerHead
            }

            getgenv().FPSBooster.config = fpsBoosterConfig

            if autoApplyBooster then
                applyFPSBooster()
            end

            if revertEnvironmentOnRestore then
                worldDescendantAddedConnection = workspaceRef.DescendantAdded:Connect(function(newInstance)
                    task.wait(0.02)

                    pcall(function()
                        if newInstance:IsA("Sky") then
                            pcall(function()
                                newInstance:Destroy()
                            end)
                        elseif newInstance:IsA("Atmosphere") then
                            pcall(function()
                                newInstance:Destroy()
                            end)

local touchSensitivityConfig = {
    MIN_SENSITIVITY = 0.1,
    MAX_SENSITIVITY = 10,
    DEFAULT_SENSITIVITY = 1,
    SAVE_FOLDER = "TouchSensitivity",
    SAVE_FILE = "settings.txt"
}

local hasFileSystemSupport = pcall(function()
    local envOk = isfolder and makefolder and isfile and readfile
    if envOk then
        envOk = writefile
    end
    return envOk
end)

local currentTouchSensitivity = touchSensitivityConfig.DEFAULT_SENSITIVITY

if hasFileSystemSupport then
    pcall(function()
        if not isfolder(touchSensitivityConfig.SAVE_FOLDER) then
            makefolder(touchSensitivityConfig.SAVE_FOLDER)
        end

        local settingsPath = touchSensitivityConfig.SAVE_FOLDER .. "/" .. touchSensitivityConfig.SAVE_FILE
        local savedValue = isfile(settingsPath) and tonumber(readfile(settingsPath))

        if savedValue then
            currentTouchSensitivity = savedValue
        end
    end)
end

local function saveTouchSensitivity(newValue)
    if hasFileSystemSupport then
        pcall(function()
            writefile(touchSensitivityConfig.SAVE_FOLDER .. "/" .. touchSensitivityConfig.SAVE_FILE, tostring(newValue))
        end)
    end
end

local function setTouchSensitivity(newValue)
    currentTouchSensitivity = newValue
    saveTouchSensitivity(newValue)
end

local function setupTouchSensitivityHooks()
    local hookApplied = false

    pcall(function()
        local localPlayer = playersService.LocalPlayer
        if localPlayer then
            local playerScripts = localPlayer:FindFirstChild("PlayerScripts")
            local playerModule = playerScripts and playerScripts:FindFirstChild("PlayerModule")

            if playerModule then
                local cameraModule = playerModule:FindFirstChild("CameraModule")
                local cameraInputModule = cameraModule and cameraModule:FindFirstChild("CameraInput")

                if cameraInputModule then
                    local success, cameraInput = pcall(function()
                        return require(cameraInputModule)
                    end)

                    if success and cameraInput and cameraInput.getRotation then
                        local originalGetRotation = cameraInput.getRotation

                        function cameraInput.getRotation(self, ...)
                            local rotationDelta = originalGetRotation(self, ...)

                            if userInputService.TouchEnabled then
                                return rotationDelta * currentTouchSensitivity
                            end

                            return rotationDelta
                        end

                        hookApplied = true
                    end
                end
            end
        end
    end)

    if not hookApplied then
        pcall(function()
            local originalIndex
            originalIndex = hookmetamethod(game, "__index", function(self, key)
                if self == userInputService and key == "MouseDelta" and userInputService.TouchEnabled then
                    return originalIndex(self, key) * currentTouchSensitivity
                end

                return originalIndex(self, key)
            end)

            hookApplied = true
        end)
    end

    return hookApplied
end

pcall(function()
    setupTouchSensitivityHooks()
end)

pcall(function()
    if mainTab and type(mainTab.CreateSlider) == "function" then
        mainTab:CreateSlider({
            Name = "Touch Sensitivity",
            Range = {
                touchSensitivityConfig.MIN_SENSITIVITY,
                touchSensitivityConfig.MAX_SENSITIVITY
            },
            Increment = 0.1,
            Suffix = "",
            CurrentValue = currentTouchSensitivity,
            Flag = "TouchSensitivity",
            Callback = function(newValue)
                setTouchSensitivity(newValue)

                if uiLibrary and type(uiLibrary.Notify) == "function" then
                    pcall(function()
                        uiLibrary:Notify({
                            Title = "Touch Sensitivity",
                            Content = ("Set to %.1f"):format(newValue),
                            Duration = 2
                        })
                    end)
                end
            end
        })
    end
end)

            local function getClosestCharacterPartToPosition(character, referencePosition)
                if not (character and referencePosition) then
                    return nil
                end

                local closestDistance = math.huge
                local closestPart = nil

                local preferredNames = {
                    "Head",
                    "UpperTorso",
                    "LowerTorso",
                    "HumanoidRootPart",
                    "LeftUpperArm",
                    "RightUpperArm",
                    "LeftLowerArm",
                    "RightLowerArm",
                    "LeftUpperLeg",
                    "RightUpperLeg",
                    "LeftLowerLeg",
                    "RightLowerLeg",
                    "LeftHand",
                    "RightHand",
                    "LeftFoot",
                    "RightFoot"
                }

                for _, partName in ipairs(preferredNames) do
                    local part = character:FindFirstChild(partName)
                    if part and part:IsA("BasePart") then
                        local distance = (part.Position - referencePosition).Magnitude
                        if distance < closestDistance then
                            closestDistance = distance
                            closestPart = part
                        end
                    end
                end

                if not closestPart then
                    for _, descendant in pairs(character:GetDescendants()) do
                        if descendant:IsA("BasePart") then
                            local distance = (descendant.Position - referencePosition).Magnitude
                            if distance < closestDistance then
                                closestDistance = distance
                                closestPart = descendant
                            end
                        end
                    end
                end

                return closestPart
            end

            local function getNearestTargetCharacter()

                local candidates = getAllValidTargets()

                        if distance < closestDistance then
                            closestDistance = distance
                            closestCharacter = character
                        end
                    end

                return closestCharacter

            local function isHumanoidInvalid(humanoid)
                if not humanoid then
                    return true
                end

                if humanoid.Health <= 0 then
                    return true
                end

                if humanoid.PlatformStand or humanoid.Sit then
                    return true
                end

                local success, state = pcall(function()
                    return humanoid:GetState()
                end)

                if success and state then
                    if state == Enum.HumanoidStateType.FallingDown then
                        return true
                    end

                    local physicsEnumSuccess = pcall(function()
                        return Enum.HumanoidStateType.Physics
                    end)

                    if physicsEnumSuccess then
                        local stateSuccess, physicsState = pcall(function()
                            return humanoid:GetState()
                        end)

                        if stateSuccess and physicsState == Enum.HumanoidStateType.Physics then
                            return true
                        end
                    end
                end

                local character = humanoid.Parent
                if character then
                    local ragdollFlag = character:FindFirstChild("Ragdoll") or character:FindFirstChild("Ragdolled")
                    if ragdollFlag and ragdollFlag:IsA("BoolValue") and ragdollFlag.Value == true then
                        return true
