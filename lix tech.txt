local function clampNumber(value, minValue, maxValue)
    if value < minValue then
        return minValue
    elseif value > maxValue then
        return maxValue
    else
        return value
    end
end

local function formatSecondsLabel(seconds)
    return string.format("%.2f", seconds):gsub("(%..-)0+$", "%1"):gsub("%.$", "") .. "s"
end

local function createTweenPulse(uiObject, targetScale, tweenInTime, tweenOutTime)
    return function()
        pcall(function()
            local originalSize = uiObject.Size
            local originalPosition = uiObject.Position
            local scaledSize = (function(size, scale)
                local xScale = size.X.Scale * scale
                local xOffset = math.floor(size.X.Offset * scale + 0.5)
                local yScale = size.Y.Scale * scale
                local yOffset = math.floor(size.Y.Offset * scale + 0.5)
                return UDim2.new(xScale, xOffset, yScale, yOffset)
            end)(originalSize, targetScale)

            local newXOffset = originalPosition.X.Offset + math.floor((originalSize.X.Offset - scaledSize.X.Offset) / 2 + 0.5)
            local newYOffset = originalPosition.Y.Offset + math.floor((originalSize.Y.Offset - scaledSize.Y.Offset) / 2 + 0.5)

            tweenService:Create(
                uiObject,
                TweenInfo.new(tweenInTime, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
                {
                    Size = scaledSize,
                    Position = UDim2.new(originalPosition.X.Scale, newXOffset, originalPosition.Y.Scale, newYOffset)
                }
            ):Play()

            task.wait(tweenInTime)

            tweenService:Create(
                uiObject,
                TweenInfo.new(tweenOutTime, Enum.EasingStyle.Back, Enum.EasingDirection.Out),
                {
                    Size = originalSize,
                    Position = originalPosition
                }
            ):Play()
        end)
    end
end

local function findNilInstanceByNameAndClass(instanceName, className)
    if not getnilinstances then
        return nil
    end

    for _, instance in pairs(getnilinstances()) do
        if instance and instance.ClassName == className and instance.Name == instanceName then
            return instance
        end
    end

    return nil
end

local character = localPlayer.Character or localPlayer.CharacterAdded:Wait()
local humanoid = character:FindFirstChildOfClass("Humanoid") or character:WaitForChild("Humanoid")
local humanoidRootPart = character:FindFirstChild("HumanoidRootPart")

local function getHumanoidStateSnapshot(targetHumanoid)
    if targetHumanoid then
        return {
            WalkSpeed = targetHumanoid.WalkSpeed,
            JumpPower = targetHumanoid.JumpPower,
            PlatformStand = targetHumanoid.PlatformStand,
            AutoRotate = targetHumanoid.AutoRotate
        }
    end

    return {
        WalkSpeed = 16,
        JumpPower = 50,
        PlatformStand = false,
        AutoRotate = true
    }
end

local humanoidDefaultState = getHumanoidStateSnapshot(humanoid)

local function onAnimationPlayed(animationTrack)
    if isLixTechEnabled and not isDashInProgress then
        if animationTrack and animationTrack.Animation then
            local success, animationAssetId = pcall(function()
                local animationIdString = animationTrack.Animation.AnimationId or ""
                return tonumber(animationIdString:match("%d+"))
            end)

            if success and animationAssetId then
                if animationAssetId == 13379003796 or animationAssetId == 10503381238 then
                    isDashInProgress = true

                    humanoid = character and character:FindFirstChildOfClass("Humanoid") or humanoid
                    humanoidRootPart = character and character:FindFirstChild("HumanoidRootPart") or humanoidRootPart

                    local currentHumanoidState = getHumanoidStateSnapshot(humanoid)

                    task.wait(dashDelaySeconds)

                    local dashCommandPayload = {
                        {
                            Dash = Enum.KeyCode.W,
                            Key = Enum.KeyCode.Q,
                            Goal = "KeyPress"
                        }
                    }

                    if character and character:FindFirstChild("Communicate") then
                        pcall(function()
                            character.Communicate:FireServer(table.unpack(dashCommandPayload))
                        end)
                    end

                    local originalBodyVelocity = findNilInstanceByNameAndClass("moveme", "BodyVelocity")
                    local originalParent = nil

                    if originalBodyVelocity then
                        originalParent = originalBodyVelocity.Parent
                        pcall(function()
                            originalBodyVelocity.Parent = nil
                        end)
                    end

                    local deleteBodyVelocityPayload = {
                        {
                            Goal = "delete bv",
                            BV = originalBodyVelocity
                        }
                    }

                    if character and character:FindFirstChild("Communicate") then
                        pcall(function()
                            character.Communicate:FireServer(table.unpack(deleteBodyVelocityPayload))
                        end)
                    end

                    task.wait(0.3)

                    if humanoidRootPart then
                        pcall(function()
                            humanoidRootPart.CFrame = humanoidRootPart.CFrame * CFrame.Angles(0, math.rad(180), 0)
                        end)
                    end

                    if originalBodyVelocity and originalBodyVelocity.Parent == nil and originalParent and originalBodyVelocity and originalBodyVelocity.Parent == nil then
                        pcall(function()
                            originalBodyVelocity.Parent = originalParent
                        end)
                    end

                    if humanoid then
                        pcall(function()
                            humanoid.WalkSpeed = currentHumanoidState.WalkSpeed or humanoidDefaultState.WalkSpeed or 16
                            humanoid.JumpPower = currentHumanoidState.JumpPower or humanoidDefaultState.JumpPower or 50
                            humanoid.PlatformStand = currentHumanoidState.PlatformStand or humanoidDefaultState.PlatformStand or false
                            humanoid.AutoRotate = currentHumanoidState.AutoRotate or humanoidDefaultState.AutoRotate or true
                        end)
                    end

                    task.wait(0.4)

                    humanoidRootPart = character and character:FindFirstChild("HumanoidRootPart") or humanoidRootPart
                    if humanoidRootPart then
                        pcall(function()
                            humanoidRootPart.CFrame = humanoidRootPart.CFrame * CFrame.Angles(0, math.rad(180), 0)
                        end)
                    end

                    task.wait(0.15)
                    isDashInProgress = false
                end
            end
        else
            return
        end
    else
        return
    end
end

local function copyToClipboard(text)
    local copied = false

    pcall(function()
        if setclipboard then
            setclipboard(text)
            copied = true
        end
    end)

    pcall(function()
        if syn and syn.set_clipboard then
            syn.set_clipboard(text)
            copied = true
        end
    end)

    pcall(function()
        if set_clipboard then
            set_clipboard(text)
            copied = true
        end
    end)

    pcall(function()
        if writeclipboard then
            writeclipboard(text)
            copied = true
        end
    end)

    return copied
end
