local function performTechFlySequence()
    humanoidRootPart.Velocity = Vector3.new(0, 200, 0)
    task.wait(0.3)

    local orbitAngle = 0
    local orbitRadius = 20
    local upwardVelocity = 40
    local orbitSpeed = 20
    local startTime = tick()
    local connection

    connection = RunService.RenderStepped:Connect(function(deltaTime)
        orbitAngle = orbitAngle + deltaTime * orbitSpeed
        local xVelocity = math.cos(orbitAngle) * orbitRadius
        local zVelocity = math.sin(orbitAngle) * orbitRadius

        humanoidRootPart.Velocity = Vector3.new(xVelocity, upwardVelocity, zVelocity)

        if tick() - startTime >= 1 then
            connection:Disconnect()
            humanoidRootPart.Velocity = Vector3.new(0, -250, 0)

            task.delay(0.2, function()
                VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.One, false, game)
                task.wait(0.1)
                VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.One, false, game)
            end)
        end
    end)
end

    VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.Two, false, game)
    task.wait(0.7)
    performTechFlySequence()
end)
